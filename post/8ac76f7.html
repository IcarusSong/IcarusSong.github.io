静态分配

c++语言顺序表定义
#include <stdio.h>
#include <stdlib.h> //malloc使用
#define MaxSize 100
typedef int ElemType;

//顺序表定义
struct SeqList    //c++中不需要使用typedef也可以直接使用SeqList声明结构体
{
    int data[MaxSize];
    int length;
    SeqList() : length(0) {} //c++中不需要init_list函数初始化
}SeqList;

//初始化
void init_list(SeqList *L)
{
    L->length = 0;
}

//尾部添加元素
int append_elem(SeqList *L, ElemType e)
{
    if (L->length >= MaxSize)
    {
        printf("list full, append error\n");
        return 0;
    }

    L->data[L->length ++ ] = e;

    return 1;
}

//遍历顺序表
void list_elem(SeqList *L)
{
    for (int i = 0; i < L->length; i ++ )
        printf("%d ", L->data[i]);

    puts("");
}

//插入元素
int insert_elem(SeqList *L, int pos, ElemType e)
{
    if (L->length >= MaxSize)
    {
        puts("list full");
        return 0;
    }

    if (pos < 1 || pos > L->length)
    {
        puts("insert position error");
        return 0;
    }

    if (pos <= L->length)
    {
        for (int i = L->length - 1; i >= pos - 1; i -- )
            L->data[i + 1] = L->data[i];

        L->data[pos - 1] = e;
        L->length ++ ;
    }

    return 1;
}

//删除元素

int delete_elem(SeqList *L, int pos, ElemType *e)
{
    if (L->length == 0)
    {
        puts("list empty");
        return 0;
    }

    if (pos < 1 || pos > L->length)
    {
        puts("delete position error");
        return 0;
    }

    *e = L->data[pos - 1];
    if (pos < L->length)
        for (int i = pos; i < L->length; i ++ )
            L->data[i - 1] = L->data[i];

    L->length -- ;

    return 1;
}

//查找元素位置

int find_elem(SeqList *L, ElemType e)
{
    if (L->length == 0)
    {
        puts("list empty");
        return 0;
    }

    for (int i = 0; i < L->length; i ++ )
        if (L->data[i] == e)
            return i + 1;

    return 0;
}

int main()
{
    SeqList L;
    init_list(&L);
    printf("L init complete, L length is : %d\n", L.length);
    printf("L storage capacity is : %d\n", sizeof L.data);

    append_elem(&L, 99);
    append_elem(&L, 66);
    list_elem(&L);

    ElemType de;
    delete_elem(&L, 1, &de);
    printf("delete elem is %d\n", de);
    list_elem(&L);

    int pos = find_elem(&L, 66);
    if (pos == 0)
    {
        printf("not find the elem\n");
    }
    else
    {
        printf("elem position is %d\n", pos);
    }

    return 0;
}

#include <iostream>
#include <cstdlib>

using namespace std;

typedef int ElemType;

class ListNode
{
public:
    ElemType data;
    ListNode *next;
    ListNode(int x) : data(x), next(nullptr) {}
};

class LinkedList
{
private:
    int length;
    ListNode *head;
public:
    LinkedList() : length(0), head(new ListNode(0)) {}

    //头插法
    void insert_head(ElemType e)
    {
        ListNode *newNode = new ListNode(e);
        newNode->next = head->next;
        head->next = newNode;
        length ++ ;
    }

    //尾插法
    void insert_tail(ElemType e)
    {
        ListNode *tail = head;
        while (tail->next)
            tail = tail->next;
        ListNode *newNode = new ListNode(e);
        tail->next = newNode;
        length ++;
    }
    //插入节点
    void insert_node(int pos, ElemType e)
    {
        if (pos < 1 || pos > length)
            puts("insert position error!");
        else
        {
            ListNode *p = head;
            int i = 0;
            while (i < pos - 1)
            {
                p = p->next;
                i ++ ;
            }

            ListNode *newNode = new ListNode(e);
            newNode->next = p->next;
            p->next = newNode;
            length ++ ;
        }
    }
    //删除节点
    void delete_node(int pos)
    {
        if (pos < 1 || pos > length)
            puts("delete position error!");
        else
        {
            ListNode *p = head;
            int i = 0;
            while (i < pos - 1)
            {
                p = p->next;
                i ++ ;
            }
            ListNode *q = p->next;
            p->next = q->next;
            delete q;
            length -- ;
        }
    }

    //获取链表长度
    int list_length()
    {
        printf("list length is %d\n", length);
        return length;
    }

    //遍历链表
    void list_node()
    {
        ListNode *curr = head->next;
        while (curr != nullptr)
        {
            printf("%d ", curr->data);
            curr = curr->next;
        }
        puts("");
    }

    //释放链表
    void free_node()
    {
        ListNode *curr = head->next;
        while (curr)
        {
            ListNode *next = curr->next;
            delete curr;
            curr = next;
        }
        head->next = nullptr;
        length = 0;
    }
};



int main()
{
    LinkedList list;
    list.insert_head(99);
    list.insert_head(88);
    list.insert_head(77);
    list.list_node();

    list.insert_tail(100);
    list.insert_tail(101);
    list.insert_tail(102);
    list.list_node();

    list.insert_node(2, 2);
    list.insert_node(5, 5);
    list.list_node();

    list.delete_node(5);
    list.delete_node(2);
    list.list_node();

    list.list_length();
    list.free_node();
    list.list_length();
}

静态链表

实现一个单链表，链表初始为空，支持三种操作：

  1. 向链表头插入一个数；
  2. 删除第 $k$ 个插入的数后面的数；
  3. 在第 $k$个插入的数后插入一个数。

现在要对该链表进行 $M$ 次操作，进行完所有操作后，从头到尾输出整个链表。

注意:题目中第 $k$ 个插入的数并不是指当前链表的第 $k$ 个数。例如操作过程中一共插入了 $k$个数，则按照插入的时间顺序，这 $k$ 个数依次为：第 $1$ 个插入的数，第 $2$ 个插入的数，…第 $n$ 个插入的数。

输入格式

第一行包含整数 $M$，表示操作次数。

接下来 MMM 行，每行包含一个操作命令，操作命令可能为以下几种：

  1. H x，表示向链表头插入一个数 $x$。
  2. D k，表示删除第 $k$ 个插入的数后面的数（当 $k$ 为 $0$ 时，表示删除头结点）。
  3. I k x，表示在第 $k$ 个插入的数后面插入一个数 $x$（此操作中 $k$ 均大于 $0$）。

输出格式

共一行，将整个链表从头到尾输出。

数据范围

$1≤M≤100000$所有操作保证合法。

输入样例：

10
H 9
I 1 1
D 1
D 0
H 6
I 3 6
I 4 5
I 4 5
I 3 4
D 6

输出样例：

6 4 6 5

#include <iostream>

using namespace std;

const int N = 100010, null = N - 1;

int head, e[N], ne[N], idx;

void init()
{
    head = 0;   //初始时，head指向空指针-1
    idx = 1;
    ne[head] = -1;
}

void insert_node(int k, int x)
{
    e[idx] = x;
    ne[idx] = ne[k];
    ne[k] = idx ++ ;
}

void delete_node(int k)
{
    ne[k] = ne[ne[k]];
}

int main()
{
    int m;
    cin >> m;

    init();

    while (m -- )
    {
        string op;
        int k, x;
        cin >> op;
        if (op == "H")
        {
            scanf("%d", &x);
            insert_node(head, x);
        }
        else if (op == "D")
        {
            scanf("%d", &k);
            delete_node(k);
        }
        else 
        {
           scanf("%d%d", &k, &x);
           insert_node(k, x);
        }
    }

    for (int i = ne[head]; i != -1; i = ne[i])
        printf("%d ", e[i]);

    puts("");

    return 0;
}
